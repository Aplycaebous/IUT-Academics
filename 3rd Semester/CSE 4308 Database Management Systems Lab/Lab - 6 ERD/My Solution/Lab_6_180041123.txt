Task 1:

Create Table Dept
(
D_name Varchar(10),
Building Varchar(10),
Primary Key (D_name)
);
Create Table Instructor
(
I_ID INT,
Name Varchar(10),
D_name Varchar(10),
Primary Key (I_ID),
Foreign Key(D_name)references Dept
);
Create Table Courses
(
C_ID INT,
Name Varchar(10),
Credit Numeric(3,2),
Primary key(C_ID)
);
Create Table Relation_Instructor_Courses
(
I_ID INT,
C_ID INT,
Primary Key(C_ID,I_ID),
Foreign Key(C_ID)references Courses,
Foreign Key(I_ID)references Instructor
);
Create Table Student
(
S_ID INT,
D_name Varchar(10),
Name Varchar(10),
Primary Key(S_ID),
Foreign Key(D_name) references Dept
);
Create Table Relation_Student_Courses
(
C_ID INT,
S_ID INT,
Primary Key(C_ID,S_ID),
Foreign Key(C_ID)references Courses,
Foreign Key(S_ID)References Student
);
Create Table Relation_Courses_Dept
(
C_ID INT,
D_name Varchar(10),
Primary Key(C_ID,D_name),
Foreign Key(C_ID)references Courses,
Foreign Key(D_name)references Dept
);

);
Create Table District
(
Dis_id INT,

Create Table Profession
(
Prof_id INT,
Name Varchar(10),

Primary Key(Prof_id)
);
Create Table Citizen
(
C_ID INT,
C_Name Varchar(10),
DOB Date,
Blood_g Varchar(10),
Div_id INT,
Prof_id INT,
Primary Key(C_ID),
Foreign Key(Div_id)references Division,

);
